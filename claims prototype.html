                          const isApproved = selectedClaim.approvals.includes(reviewer);

                          const isRejected = selectedClaim.rejections.includes(reviewer);

                          const isCurrent = selectedClaim.currentReviewer === reviewer;


                          return (

                            <div key={idx} className="flex items-center space-x-4 bg-white p-4 rounded-xl shadow-sm">

                              <div className={`w-10 h-10 rounded-xl flex items-center justify-center shadow-md ${

                                isApproved ? 'bg-gradient-to-br from-emerald-500 to-emerald-600' :

                                isRejected ? 'bg-gradient-to-br from-rose-500 to-rose-600' :

                                isCurrent ? 'bg-gradient-to-br from-blue-500 to-blue-600' :

                                'bg-slate-300'

                              }`}>

                                {isApproved ? <Check className="w-6 h-6 text-white" /> :

                                 isRejected ? <X className="w-6 h-6 text-white" /> :

                                 isCurrent ? <Clock className="w-6 h-6 text-white" /> :

                                 <span className="text-white font-bold">{idx + 1}</span>}

                              </div>

                              <div className="flex-1">

                                <p className="font-bold text-slate-800">{reviewer} Team</p>

                                <p className="text-sm text-slate-600 font-medium">

                                  {isApproved ? 'Approved' : isRejected ? 'Rejected' : isCurrent ? 'Currently reviewing' : 'Pending'}

                                </p>

                              </div>

                            </div>

                          );

                        })}

                      </div>

                    </div>


                    {selectedClaim.comments.length > 0 && (

                      <div>

                        <h3 className="font-bold text-slate-800 mb-4 text-lg">Comments and Feedback</h3>

                        <div className="space-y-3">

                          {selectedClaim.comments.map((c, idx) => (

                            <div key={idx} className="bg-amber-50 border border-amber-200 rounded-xl p-5">

                              <div className="flex items-center justify-between mb-3">

                                <span className="font-bold text-slate-800">{c.reviewer} Team</span>

                                <span className="text-sm text-slate-600 font-medium">{c.date}</span>

                              </div>

                              <p className="text-slate-700">{c.text}</p>

                            </div>

                          ))}

                        </div>

                      </div>

                    )}


                    {selectedClaim.currentReviewer === userRole && selectedClaim.status !== 'approved' && selectedClaim.status !== 'rejected' && (

                      <div className="flex space-x-4 pt-6 border-t border-slate-200">

                        <button

                          onClick={() => handleApprove(selectedClaim.id)}

                          className="flex-1 bg-gradient-to-r from-emerald-600 to-emerald-700 hover:from-emerald-700 hover:to-emerald-800 text-white font-bold py-4 rounded-xl flex items-center justify-center space-x-2 transition-all shadow-lg hover:shadow-xl"

                        >

                          <Check className="w-6 h-6" />

                          <span>Approve Claim</span>

                        </button>

                        <button

                          onClick={() => handleReject(selectedClaim.id)}

                          className="flex-1 bg-gradient-to-r from-rose-600 to-rose-700 hover:from-rose-700 hover:to-rose-800 text-white font-bold py-4 rounded-xl flex items-center justify-center space-x-2 transition-all shadow-lg hover:shadow-xl"

                        >

                          <X className="w-6 h-6" />

                          <span>Reject Claim</span>

                        </button>

                      </div>

                    )}

                  </div>

                </div>

              ) : (

                <div className="grid gap-5">

                  {filteredClaims.length === 0 ? (

                    <div className="text-center py-16 bg-white rounded-2xl shadow-sm border border-slate-200">

                      <FileText className="w-20 h-20 text-slate-300 mx-auto mb-6" />

                      <h3 className="text-2xl font-bold text-slate-700 mb-3">No claims found</h3>

                      <p className="text-slate-500 text-lg">

                        {currentPage === 'dashboard'

                          ? 'You have no pending claims to review at this time.'

                          : 'No claims match your search criteria.'}

                      </p>

                    </div>

                  ) : (

                    filteredClaims.map(claim => (

                      <div

                        key={claim.id}

                        onClick={() => setSelectedClaim(claim)}

                        className="bg-white rounded-2xl shadow-md hover:shadow-xl transition-all cursor-pointer p-6 border-2 border-slate-200 hover:border-blue-400"

                      >

                        <div className="flex items-start justify-between mb-5">

                          <div className="flex-1">

                            <h3 className="text-xl font-bold text-slate-800 mb-3">{claim.title}</h3>

                            <p className="text-slate-600 line-clamp-2 leading-relaxed">{claim.description}</p>

                          </div>

                          <span className={`px-4 py-2 rounded-xl text-xs font-bold flex items-center space-x-2 ml-4 ${getStatusColor(claim.status)}`}>

                            {getStatusIcon(claim.status)}

                            <span className="ml-1 capitalize">{claim.status.replace('_', ' ')}</span>

                          </span>

                        </div>


                        <div className="flex items-center justify-between">

                          <div className="flex items-center space-x-5 text-sm text-slate-600">

                            <span className="flex items-center font-medium">

                              <Users className="w-4 h-4 mr-2" />

                              {claim.submittedBy}

                            </span>

                            <span className="flex items-center font-medium">

                              <Clock className="w-4 h-4 mr-2" />

                              {claim.submissionDate}

                            </span>

                            <span className="bg-blue-100 text-blue-700 px-3 py-1 rounded-lg font-semibold border border-blue-200">

                              {claim.offering}

                            </span>

                          </div>


                          <div className="flex items-center space-x-3">

                            <div className="flex -space-x-2">

                              {claim.approvals.map((approver, idx) => (

                                <div key={idx} className="w-8 h-8 bg-gradient-to-br from-emerald-500 to-emerald-600 rounded-full flex items-center justify-center border-2 border-white shadow-md">

                                  <Check className="w-4 h-4 text-white" />

                                </div>

                              ))}

                            </div>

                            <span className="text-slate-700 font-bold">

                              {claim.approvals.length}/{claim.reviewers.length}

                            </span>

                          </div>

                        </div>

                      </div>

                    ))

                  )}

                </div>

              )}

            </div>

          </div>


          {showCommentModal && (

            <div className="fixed inset-0 bg-black bg-opacity-60 flex items-center justify-center p-4 z-50 backdrop-blur-sm">

              <div className="bg-white rounded-2xl shadow-2xl max-w-lg w-full p-8 border border-slate-200">

                <h2 className="text-2xl font-bold text-slate-800 mb-4">Rejection Reason</h2>

                <p className="text-slate-600 mb-5">Please provide a detailed reason for rejecting this claim:</p>

                <textarea

                  value={comment}

                  onChange={(e) => setComment(e.target.value)}

                  className="w-full border-2 border-slate-200 rounded-xl p-4 h-36 focus:ring-2 focus:ring-rose-500 focus:border-rose-500 transition-all"

                  placeholder="Enter your comments and feedback..."

                />

                <div className="flex space-x-3 mt-6">

                  <button

                    onClick={submitRejection}

                    className="flex-1 bg-gradient-to-r from-rose-600 to-rose-700 hover:from-rose-700 hover:to-rose-800 text-white font-bold py-3 rounded-xl transition-all shadow-lg"

                  >

                    Submit Rejection

                  </button>

                  <button

                    onClick={() => {

                      setShowCommentModal(false);

                      setComment('');

                    }}

                    className="flex-1 bg-slate-200 hover:bg-slate-300 text-slate-800 font-bold py-3 rounded-xl transition-all"

                  >

                    Cancel

                  </button>

                </div>

              </div>

            </div>

          )}

        </div>

      );

    }


    const root = ReactDOM.createRoot(document.getElementById('root'));

    root.render(<ClaimsApprovalPrototype />);
