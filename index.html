<script>

  const claimsData = [

    {

      id: 1,

      theme: "Product",

      offering: "Synchrony",

      type: "Competitive",

      claim: "Synchrony is the only technology that uses image guidance to automatically adapt and synchronize the radiation beam to the position."

    },

    {

      id: 2,

      theme: "Software",

      offering: "Precision",

      type: "Clinical",

      claim: "Deliver more personalized and more precise treatment plans for more patients."

    },

    {

      id: 3,

      theme: "Software",

      offering: "ClearRT",

      type: "Competitive",

      claim: "ClearRT avoids the distortions caused by MR imaging and the artifacts associated with conventional cone beam CT by using hardware."

    },

    {

      id: 4,

      theme: "Software",

      offering: "ClearRT",

      type: "Technological",

      claim: "It’s able to image a meter in a minute, it provides clinicians the industry’s longest continuous scan length, up to 135 cm, for fast and accurate assessment."

    },

    {

      id: 5,

      theme: "Software",

      offering: "ClearRT",

      type: "Competitive",

      claim: "The industry leading image quality and anatomical accuracy produced by ClearRT allows clinicians to quickly and confidently setup treatment."

    }

  ];

 

  let approvedClaims = [];

 

  function showSection(sectionId) {

    document.querySelectorAll('main > section').forEach(section => {

      section.classList.add('hidden');

    });

    document.getElementById(sectionId).classList.remove('hidden');

  }

 

  function loadPendingClaims() {

    const role = document.getElementById('roleSelect').value;

    const tbody = document.querySelector('#pendingClaimsTable tbody');

    tbody.innerHTML = '';

 

    const filtered = claimsData.filter(claim =>

      (role === "product" && claim.theme.toLowerCase() === "product") ||

      (role === "regulatory" && (claim.type.toLowerCase() === "competitive" || claim.type.toLowerCase() === "technological")) ||

      (role === "clinical" && claim.type.toLowerCase() === "clinical") ||

      (role === "engineering" && claim.theme.toLowerCase() === "software")

    );

 

    filtered.forEach((claim, index) => {

      const tr = document.createElement('tr');

      tr.innerHTML = `

        <td>${index + 1}</td>

        <td>${claim.theme}</td>

        <td>${claim.offering}</td>

        <td>${claim.type}</td>

        <td>${claim.claim.substring(0, 50)}...</td>

        <td><button onclick="reviewClaim(${claim.id})">Review</button></td>

      `;

      tbody.appendChild(tr);

    });

  }

 

  function reviewClaim(id) {

    const claim = claimsData.find(c => c.id === id);

    if (!claim) return;

    document.getElementById('claimTheme').textContent = claim.theme;

    document.getElementById('claimOffering').textContent = claim.offering;

    document.getElementById('claimType').textContent = claim.type;

    document.getElementById('claimText').textContent = claim.claim;

    showSection('claimReview');

    currentReviewClaimId = id;

  }

 

  let currentReviewClaimId = null;

 

  function approveClaim() {

    if (currentReviewClaimId == null) return alert('No claim selected for approval.');

    const claim = claimsData.find(c => c.id === currentReviewClaimId);

    approvedClaims.push(claim);

    alert('Claim approved!');

    showSection('home');

    currentReviewClaimId = null;

    loadPendingClaims();

    loadApprovedClaims();

  }

 

  function rejectClaim() {

    if (currentReviewClaimId == null) return alert('No claim selected for rejection.');

    alert('Claim rejected.');

    showSection('home');

    currentReviewClaimId = null;

    loadPendingClaims();

  }

 

  function loadApprovedClaims() {

    const tbody = document.querySelector('#approvedClaimsTable tbody');

    tbody.innerHTML = '';

    approvedClaims.forEach((claim, index) => {

      const tr = document.createElement('tr');

      tr.innerHTML = `

        <td>${index + 1}</td>

        <td>${claim.theme}</td>

        <td>${claim.offering}</td>

        <td>${claim.type}</td>

        <td>${claim.claim}</td>

      `;

      tbody.appendChild(tr);

    });

  }

 

  // Initialize

  loadPendingClaims();

</script>
